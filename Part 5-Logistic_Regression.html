# 1
df = pd.read_csv('Titanic.csv')
print(df.head(5))
# 2 Survived is the target variable
# 3 Obviously, variable Passenger is not likely to be relevant for logistic regression.
df = df.drop(columns=['Passenger'])
# 4 There are no missing values.
print(df.isnull().sum())
# 5
fig, axs = plt.subplots(2, 2, figsize=(10, 10))
plt.suptitle('Count Plots of Sex Class and Age')
countplot(ax=axs[0][0], data=df, x='Sex')
countplot(ax=axs[0][1], data=df, x='Class')
countplot(ax=axs[1][0], data=df, x='Age')
countplot(ax=axs[1][1], data=df, x='Survived')
plt.show()
# 6
df = pd.get_dummies(df, columns=['Sex', 'Class', 'Age'])
# 7
X = df.drop(columns=['Survived'])
y = df['Survived']
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=2020)
# 8
logreg = LogisticRegression()
logreg.fit(X_train, y_train)
# 9
prediction = logreg.predict(X_test)
# 10
print(classification_report(y_test, prediction))
# 11
plot_confusion_matrix(logreg, X_test, y_test)
plt.show()
